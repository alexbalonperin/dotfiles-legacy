          def self.from_rgb_triple(rgb_triple)
            r, g, b = rgb_triple.red, rgb_triple.green, rgb_triple.blue
            y = (0.299 * r + 0.587 * g + 0.114 * b).round
            u = ((b - y) * 0.492).round
            v = ((r - y) * 0.877).round
            new(y, u, v)
          end
