    def scene=(n)
        if n.nil?
            Kernel.raise IndexError, "scene number out of bounds" unless @images.length == 0
            @scene = nil
            return @scene
        elsif @images.length == 0
            Kernel.raise IndexError, "scene number out of bounds"
        end

        n = Integer(n)
        if n < 0 || n > length - 1
            Kernel.raise IndexError, "scene number out of bounds"
        end
        @scene = n
        return @scene
    end
