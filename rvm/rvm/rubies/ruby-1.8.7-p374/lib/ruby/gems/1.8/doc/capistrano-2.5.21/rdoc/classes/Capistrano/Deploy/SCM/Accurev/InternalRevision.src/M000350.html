          def self.parse(string)
            match = /([^\/]+)(\/(.+)){0,1}/.match(string)
            raise "Unrecognized revision identifier: #{string}" unless match

            stream = match[1]
            transaction_id = match[3] || 'highest'
            InternalRevision.new(stream, transaction_id)
          end
