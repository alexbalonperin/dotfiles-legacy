    def []=(*args)
        obj = @images.[]=(*args)
        if obj && obj.respond_to?(:each) then
            is_an_image_array(obj)
            set_current obj.last.__id__
        elsif obj
            is_an_image(obj)
            set_current obj.__id__
        else
            set_current nil
        end
        return obj
    end
