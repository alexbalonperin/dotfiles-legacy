                def text_rel_coords(text)
                    y_rel_coords = []
                    x_rel_coords = []
                    first_word = true
                    words = text.split(::Magick::RVG::WORD_SEP)
                    words.each do |word|
                        unless first_word
                            wx, wy = get_word_spacing()
                            x_rel_coords << wx
                            y_rel_coords << wy
                        end
                        first_word = false
                        word.split('').each do |glyph|
                            wx, wy = get_letter_spacing(glyph)
                            x_rel_coords << wx
                            y_rel_coords << wy
                        end
                    end
                    [x_rel_coords, y_rel_coords]
                end
