      def within(directory, &block)
        (@pwd ||= []).push directory.to_s
        execute "if test ! -d \#{File.join(@pwd)}\nthen echo \"Directory does not exist '\#{File.join(@pwd)}'\" 1>&2\nfalse\nfi\n", verbosity: Logger::DEBUG
          yield
      ensure
        @pwd.pop
      end
