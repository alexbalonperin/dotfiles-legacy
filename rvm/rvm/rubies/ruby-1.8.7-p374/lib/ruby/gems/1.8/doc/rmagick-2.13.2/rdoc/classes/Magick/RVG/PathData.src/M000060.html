            def curveto(abs, x1, y1, x2, y2, x, y, *coords)
                @path << sprintf("%s%g,%g %g,%g %g,%g ", (abs ? 'C' : 'c'), x1, y1, x2, y2, x, y)
                # "multiple sets of coordinates may be specified to draw a polybezier"
                add_points(6, *coords)
            end
