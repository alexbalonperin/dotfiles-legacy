    def subhash(*patterns)
      patterns.map! do |pat|
        pat = pat.to_sym.to_s if pat.respond_to?(:to_sym)
        pat.respond_to?(:match) ? pat : pat.to_s
      end
      result =
        if default_proc
          self.class.new(&default_proc)
        else
          self.class.new(default)
        end
      if block_given?
        each do |k, v|
          patterns.each { |pat|
            if pat === k.to_s
              result[k] = yield(k, v, $~)
              break
            end
          }
        end
      else
        each do |k, v|
          result[k] = v if patterns.any? { |pat| pat === k.to_s }
        end
      end
      result
    end
