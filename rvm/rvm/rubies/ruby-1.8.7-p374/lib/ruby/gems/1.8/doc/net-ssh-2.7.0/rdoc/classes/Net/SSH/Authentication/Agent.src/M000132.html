    def sign(key, data)
      signed = @agent_proxy.sign(key.java_blob, data.to_java_bytes)
      String.from_java_bytes(signed)
    rescue AgentProxyException => e
      raise AgentError,
        "agent could not sign data with requested identity: #{e.message}",
        e.backtrace
    end
