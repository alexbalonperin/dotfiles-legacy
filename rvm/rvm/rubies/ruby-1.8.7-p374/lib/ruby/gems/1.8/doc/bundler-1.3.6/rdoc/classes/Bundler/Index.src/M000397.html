    def unmet_dependency_names
      dependency_names = specs.values.map do |array_of_s|
        array_of_s.map do |s|
          s.dependencies.map{|d| d.name }
        end
      end.flatten.uniq
      dependency_names.select{|name| name != 'bundler' && specs_by_name(name).empty? }
    end
