        def self.convert_to_float(*args)
            allow_nil = false
            if args.last == :allow_nil
                allow_nil = true
                args.pop
            end
            begin
                fargs = args.collect { |a| (allow_nil && a.nil?) ? a : Float(a) }
            rescue ArgumentError, TypeError
                raise ArgumentError, self.fmsg(*args)
            end
            return fargs
        end
