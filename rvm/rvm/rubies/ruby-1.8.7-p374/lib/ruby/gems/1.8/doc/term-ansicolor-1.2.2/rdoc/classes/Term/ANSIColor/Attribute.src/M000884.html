      def self.nearest_rgb_color(color, options = {})
        rgb = RGBTriple[color]
        colors = rgb_colors
        if options.key?(:gray) && !options[:gray]
          colors = colors.reject(&:gray?)
        end
        colors.reject(&:background?).min_by { |c| c.distance_to(rgb, options) }
      end
