            def rvg(cols, rows, x=0, y=0, &block)
                rvg = Magick::RVG.new(cols, rows, &block)
                begin
                    x, y = Float(x), Float(y)
                rescue ArgumentError
                    args = [cols, rows, x, y]
                    raise ArgumentError, "at least one argument is not convertable to Float (got #{args.collect {|a| a.class}.join(', ')})"
                end
                rvg.corner(x, y)
                @content << rvg
                return rvg
            end
