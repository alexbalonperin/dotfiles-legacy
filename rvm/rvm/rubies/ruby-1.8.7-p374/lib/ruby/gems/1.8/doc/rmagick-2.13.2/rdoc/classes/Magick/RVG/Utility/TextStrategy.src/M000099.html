                def enquote(text)
                    if text.length > 2 && /\A(?:\"[^\"]+\"|\'[^\']+\'|\{[^\}]+\})\z/.match(text)
                        return text
                    elsif !text['\'']
                        text = '\''+text+'\''
                        return text
                    elsif !text['"']
                        text = '"'+text+'"'
                        return text
                    elsif !(text['{'] || text['}'])
                        text = '{'+text+'}'
                        return text
                    end

                    # escape existing braces, surround with braces
                    text.gsub!(/[}]/) { |b| '\\' + b }
                    return '{' +  text + '}'
                end
