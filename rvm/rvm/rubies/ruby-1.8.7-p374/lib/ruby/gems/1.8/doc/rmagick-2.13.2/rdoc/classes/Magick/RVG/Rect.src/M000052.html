            def round(rx, ry=nil)
                rx, ry = Magick::RVG.convert_to_float(rx, ry || rx)
                if rx < 0 || ry < 0
                    raise ArgumentError, "rx, ry must be >= 0 (#{rx}, #{ry} given)"
                end
                @args << rx << ry
                @primitive = :roundrectangle
                self
            end
