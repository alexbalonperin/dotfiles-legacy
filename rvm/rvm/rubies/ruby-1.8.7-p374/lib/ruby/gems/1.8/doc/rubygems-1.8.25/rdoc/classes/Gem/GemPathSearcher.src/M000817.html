  def find_in_unresolved(glob)
    # HACK violation
    specs = Gem.unresolved_deps.values.map { |dep|
      Gem.source_index.search dep, true
    }.flatten

    specs.select do |spec|
      # TODO: inverted responsibility
      matching_file? spec, glob
    end || []
  end
