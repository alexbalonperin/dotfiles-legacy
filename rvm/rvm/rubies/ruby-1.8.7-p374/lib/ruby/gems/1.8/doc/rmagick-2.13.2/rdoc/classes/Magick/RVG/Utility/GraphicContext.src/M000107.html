                def affine(sx, rx, ry, sy, tx, ty)
                    sx, rx, ry, sy, tx, ty = Magick::RVG.convert_to_float(sx, rx, ry, sy, tx, ty)
                    @gc.affine(sx, rx, ry, sy, tx, ty)
                    @text_attrs.set_affine(sx, rx, ry, sy, tx, ty)
                    nil
                end
