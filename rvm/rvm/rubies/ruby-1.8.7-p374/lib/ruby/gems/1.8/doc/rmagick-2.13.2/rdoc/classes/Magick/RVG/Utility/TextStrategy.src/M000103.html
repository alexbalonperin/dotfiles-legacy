                def render_glyph(glyph_orientation, x, y, glyph)
                    if glyph_orientation == 0
                        @ctx.gc.text(x, y, enquote(glyph))
                    else
                        @ctx.gc.push
                        @ctx.gc.translate(x, y)
                        @ctx.gc.rotate(glyph_orientation)
                        @ctx.gc.translate(-x, -y)
                        @ctx.gc.text(x, y, enquote(glyph))
                        @ctx.gc.pop
                    end
                end
