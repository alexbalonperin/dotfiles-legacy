    def example_group_finished(notification)
      @group_nesting -= 1
      if @group_nesting == 0
        path = (RSPEC_3 ? notification.group.file_path : notification.file_path)
        @example_times[path] += ParallelTests.now - @time
      end
      super if defined?(super)
    end
