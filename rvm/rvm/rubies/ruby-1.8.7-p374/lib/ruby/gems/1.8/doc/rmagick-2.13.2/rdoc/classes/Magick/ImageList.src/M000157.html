    def *(n)
        unless n.kind_of? Integer
            Kernel.raise ArgumentError, "Integer required (#{n.class} given)"
        end
        current = get_current()
        ilist = self.class.new
        (@images * n).each {|image| ilist << image}
        ilist.set_current current
        return ilist
    end
