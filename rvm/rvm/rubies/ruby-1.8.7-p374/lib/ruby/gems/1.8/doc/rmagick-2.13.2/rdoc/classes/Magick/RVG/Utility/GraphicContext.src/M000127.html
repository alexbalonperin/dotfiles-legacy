                def text(x, y, text)
                    return if text.length == 0
                    if @text_attrs.non_default?
                        text_renderer = TEXT_STRATEGIES[@text_attrs.writing_mode].new(self)
                    else
                        text_renderer = DefaultTextStrategy.new(self)
                    end

                    return text_renderer.render(x, y, text)
                end
