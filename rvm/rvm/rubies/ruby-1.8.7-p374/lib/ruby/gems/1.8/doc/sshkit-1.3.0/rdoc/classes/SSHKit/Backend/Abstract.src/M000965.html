      def as(who, &block)
        if who.is_a? Hash
          @user  = who[:user]  || who["user"]
          @group = who[:group] || who["group"]
        else
          @user  = who
          @group = nil
        end
        execute "if ! sudo su \#{@user} -c whoami > /dev/null\nthen echo \"You cannot switch to user '\#{@user}' using sudo, please check the sudoers file\" 1>&2\nfalse\nfi\n", verbosity: Logger::DEBUG
        yield
      ensure
        remove_instance_variable(:@user)
        remove_instance_variable(:@group)
      end
