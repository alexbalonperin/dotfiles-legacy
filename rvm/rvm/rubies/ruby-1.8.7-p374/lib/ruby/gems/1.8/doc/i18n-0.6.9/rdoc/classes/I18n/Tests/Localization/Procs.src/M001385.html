          def inspect_args(args)
            args = args.map do |arg|
              case arg
              when ::Time, ::DateTime
                arg.strftime('%a, %d %b %Y %H:%M:%S %Z').sub('+0000', '+00:00')
              when ::Date
                arg.strftime('%a, %d %b %Y')
              when Hash
                arg.delete(:fallback)
                arg.inspect
              else
                arg.inspect
              end
            end
            "[#{args.join(', ')}]"
          end
