    def initialize(h={})
      @all = [self]

      @street_address=h[:street_address]
      @sub_premise=nil
      @street_number=nil
      @street_name=nil
      @city=h[:city]
      @state=h[:state]
      @zip=h[:zip]
      @country_code=h[:country_code]
      @province = h[:province]
      @success=false
      @precision='unknown'
      @full_address=nil
      super(h[:lat],h[:lng])
    end
