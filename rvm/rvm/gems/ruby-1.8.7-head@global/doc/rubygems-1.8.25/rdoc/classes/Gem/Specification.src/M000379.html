  def conflicts
    conflicts = {}
    Gem.loaded_specs.values.each do |spec|
      bad = self.runtime_dependencies.find_all { |dep|
        spec.name == dep.name and not spec.satisfies_requirement? dep
      }

      conflicts[spec] = bad unless bad.empty?
    end
    conflicts
  end
