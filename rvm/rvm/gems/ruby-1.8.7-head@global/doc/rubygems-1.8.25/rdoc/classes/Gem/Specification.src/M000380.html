  def contains_requirable_file? file
    root = full_gem_path

    require_paths.each do |lib|
      base = "#{root}/#{lib}/#{file}"
      Gem.suffixes.each do |suf|
        path = "#{base}#{suf}"
        return true if File.file? path
      end
    end

    return false
  end
